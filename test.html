Fixed Navigation with Active Link Indicator

Fixed Issues:

  • Multiple Active Links: Only one link is active at a time now
  • Home Page Detection: Added proper home page detection
  • Indicator Positioning: Fixed calculation for accurate positioning
  • Smooth Transitions: Added smooth animation for the indicator
  • Resize Handling: Improved resize event handling

Try clicking the navigation links to see the indicator in action!

Fixed JavaScript:

const links = document.querySelectorAll(".nav-link");
const indicator = document.querySelector(".nav-indicator");

// Initialize the indicator
function initIndicator() {
    const activeLink = document.querySelector(".nav-link.active");
    if (activeLink) {
        setIndicator(activeLink);
    } else if (links.length > 0) {
        // Set first link as active if none is active
        links[0].classList.add("active");
        setIndicator(links[0]);
    }
}

// Set indicator position and width
function setIndicator(el) {
    const rect = el.getBoundingClientRect();
    const navList = el.closest(".header__nav-list");
    const navRect = navList.getBoundingClientRect();
    
    indicator.style.width = `${rect.width}px`;
    indicator.style.left = `${rect.left - navRect.left}px`;
}

// Set initial indicator position
window.addEventListener("DOMContentLoaded", () => {
    // Detect home page
    const path = window.location.pathname;
    const isHomePage = path.endsWith("/") || path.endsWith("index.html") || path === "";
    
    // Remove active class from all links
    links.forEach(link => link.classList.remove("active"));
    
    let activeLink = null;
    
    // Find matching link
    links.forEach(link => {
        const page = link.getAttribute("data-page");
        
        // Special handling for home page
        if (isHomePage && page === "home") {
            activeLink = link;
        }
        // For other pages
        else if (path.includes(page)) {
            activeLink = link;
        }
    });
    
    // Set active link
    if (activeLink) {
        activeLink.classList.add("active");
    } else if (links.length > 0) {
        // Fallback to first link
        links[0].classList.add("active");
    }
    
    initIndicator();
});

// Update indicator on link click
links.forEach((link) => {
    link.addEventListener("click", (e) => {
        e.preventDefault();
        
        // Remove active class from all links
        links.forEach((l) => l.classList.remove("active"));
        
        // Add active class to clicked link
        link.classList.add("active");
        
        // Update indicator position
        setIndicator(link);
    });
});

// Update indicator on window resize
window.addEventListener("resize", () => {
    const active = document.querySelector(".nav-link.active");
    if (active) setIndicator(active);
});
